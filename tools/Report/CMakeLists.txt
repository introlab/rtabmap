
set(LIBRARIES rtabmap_core)

IF(WITH_QT AND (QT4_FOUND OR Qt5_FOUND OR Qt6_FOUND))
	set(LIBRARIES ${LIBRARIES} rtabmap_gui)
ENDIF()

ADD_EXECUTABLE(report main.cpp)

TARGET_LINK_LIBRARIES(report ${LIBRARIES})

SET_TARGET_PROPERTIES(report PROPERTIES
	OUTPUT_NAME ${PROJECT_PREFIX}-report
)

IF(WITH_QT AND (QT4_FOUND OR Qt5_FOUND OR Qt6_FOUND))
	target_compile_definitions(report PRIVATE WITH_QT)
ENDIF()

INSTALL(TARGETS report
	RUNTIME DESTINATION "${CMAKE_INSTALL_BINDIR}" COMPONENT runtime
	BUNDLE DESTINATION "${CMAKE_BUNDLE_LOCATION}" COMPONENT runtime)
